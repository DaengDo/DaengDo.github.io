```c
#include <stdio.h>
#include <float.h>

int main(void)
{
    // round-off errors (ex1) 번역으로는 '빈올림오류' 라고 하지만 적당한 번역은 아닌듯...
    
    float a, b;
    a = 1.0E20f + 1.0f;
    b = a - 1.0E20f;
    printf("%f\n", b);

    // round-off errors (ex2)
    
    float a = 0.0f;
    a = a + 0.01f;          // 0.01 에 마우스를 갖다대면 정확히 0.01이 아닌것을 확인할 수 있다.

    // overflow

    float max = 3.402823466e+38F;
    printf("%f\n", max);
    max = max * 100.0f;
    printf("%f\n", max);

    // underflow

    float f = 1.401298464e-45F;
    printf("%e\n", f);
    f = f / 2.0f; // subnormal 부동소수점으로는 더 이상 표현 할 수 없는 작은 값
    printf("%e\n", f);

    // 0 divide error 0으로 나뉘지않도록 알고리듬을 잘 짜야한다.

    float f = 104.0f;
    printf("%e\n", f);
    f = f / 0.0f;
    printf("%e\n", f);

    return 0;
}

```
