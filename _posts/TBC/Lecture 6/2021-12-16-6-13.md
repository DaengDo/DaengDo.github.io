- do while 문은 최초 한번의 루프를 보장하고 싶을 때 사용한다.
- 순서도를 통해 한번의 루프가 보장됨을 살펴볼 수 있다.
  - <img src="https://user-images.githubusercontent.com/82828382/146296064-cb943964-e436-4f1b-b000-385b3127522c.png">

```c
#include <stdio.h>

int main()
{
  const int key = 123;
  int password = 0;

  // printf("Enter secret code : ");
  // scanf("%d", &password);

  // while (password != key)
  // {
  //   printf("Enter secret code : ");
  //   scanf("%d", &password);
  // }

  do
  {
    printf("Enter secret code : ");
    scanf("%d", &password);
  }
  while (password != key);

  printf("confirmed");

  return 0;
}
```
