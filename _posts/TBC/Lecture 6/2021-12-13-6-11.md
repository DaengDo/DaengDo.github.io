```c
#include <stdio.h>

int main()
{
  for (int n = 1, nsqr = n * n; n < 10; n++, nsqr = n * n)
    printf("%d %d\n", n, nsqr);

  // printf() 안의 콤마는 연산자가 아니라 구분지어주는 역할을 할 뿐이다.

  int i, j;
  i = 1;
  i++, j = i; 
  // sequence point 역할을 하므로 i는 콤마를 만나는 순간2가 됨
  printf("%d %d\n", i, j);

  int x, y, z;
  // `=` 연산자의 결합법칙 순서는 right to left
  z = x = 1, y = 2;
  printf("x=%d, y=%d, z=%d", x, y, z);
  z = (x = 1), (y = 2);
  printf("x=%d, y=%d, z=%d", x, y, z);
  
  z = ((x = 1), (y = 2));
  // `,` 연산자는 left to right 따라서 z 의 값은 2가 된다
  printf("x=%d, y=%d, z=%d", x, y, z);


  return 0;
}
```
